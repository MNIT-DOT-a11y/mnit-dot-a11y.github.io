Instructions to Install and Run Axe-Core Selenium scripts with Java

This guide provides step-by-step instructions for setting up your environment and running an accessibility scan on a web page using Java, Axe-core, and Selenium.

1. Download and Install Java Development Kit (JDK)

You need the JDK to compile and run Java applications.

  • Download JDK: Go to the official Oracle JDK download page (or choose an OpenJDK distribution like Adoptium Temurin). Choose the Windows installer (.exe) that matches your system (usually x64).
  • Run the Installer: Double-click the downloaded `.exe` file and follow the on-screen instructions. The default installation path (e.g., `C:\Program Files\Java\jdk-[version]`) is usually fine.
  • Set JAVA_HOME Environment Variable:
  1. Right-click the Start button and select System.
  2. Click Advanced system settings.
  3. In the System Properties window, click Edit the System Environment Variables for your account.
  4. Under System variables, click New....
  5. For Variable name, enter `JAVA_HOME`.
  6. For Variable value, enter the path to your JDK installation directory (e.g., `C:\Program Files\Java\jdk-17`).
  7. Click OK.
  • Update the Path Variable:
  1. In the System variables section, find the `Path` variable and click Edit....
  2. Click New and add `%JAVA_HOME%\bin`.
  3. Make sure this entry is moved up to a high priority to ensure the correct JDK is used.
  4. Click OK on all open windows.
  • Verify Installation: Open a new Command Prompt or PowerShell window and type `java -version` and `javac -version`. You should see the installed JDK version information.

2. Download and Install Apache Maven

Maven is a build automation tool used for Java projects.

  • Download Maven: Go to the Apache Maven download page. Download the Binary zip archive (e.g., `apache-maven-3.9.6-bin.zip`).
  • Extract Maven: Unzip the downloaded archive to a location like `C:\Program Files\Apache\apache-maven-[version]`.
  • Set MAVEN_HOME (or M2_HOME) Environment Variable: Follow the same steps as for JAVA_HOME.
  1. For Variable name, enter `MAVEN_HOME` (or `M2_HOME`).
  2. For Variable value, enter the path to your Maven installation directory (e.g., `C:\Program Files\Apache\apache-maven-3.9.6`).
  3. Click OK.
  • Update the Path Variable:
  1. In the System variables section, find the `Path` variable and click Edit....
  2. Click New and add `%MAVEN_HOME%\bin`.
  3. Click OK on all open windows.
  • Verify Installation: Open a new Command Prompt or PowerShell window and type `mvn -version`. You should see the Maven version information.

3. Clone Axe-core Selenium Repository

You'll download the axe-core-maven-html repository using Git.

  • Install Git: If you don't have Git installed, download it from git-scm.com and follow the installation instructions.
  • Clone the Repository: Open your terminal and navigate to the directory where you want to save the repository (e.g., `C:\Users\rubi1dou\Documents`).
git clone https://github.com/dequelabs/axe-core-maven-html.git

This will create a new folder named `axe-core-maven-html` containing the repository's contents.

4. Create and Configure LiveAxeCoreTest.java

This is the test file where you will write the accessibility scan logic.

  • Create the file: Inside the cloned repository, navigate to `axe-core-maven-html/selenium/src/test/java/com/yourcompany/axtest/`. Create a new file named `LiveAxeCoreTest.java` (if the directories don't exist, create them).
  • Add the code: Copy and paste the provided JUnit 4 code into the new file. This code uses the `Parameterized` runner to test multiple sites.
package com.yourcompany.axtest;

import com.deque.html.axecore.results.AxeRuntimeException;
import com.deque.html.axecore.results.Results;
import com.deque.html.axecore.results.Rule;
import com.deque.html.axecore.results.Node;
import com.deque.html.axecore.selenium.AxeBuilder;
import io.github.bonigarcia.wdm.WebDriverManager;
import org.junit.After;
import org.junit.Before;
import org.junit.Test;
import org.junit.Assert;
import org.junit.runner.RunWith;
import org.junit.runners.Parameterized;
import org.junit.runners.Parameterized.Parameters;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.chrome.ChromeDriver;
import org.openqa.selenium.chrome.ChromeOptions;
import com.fasterxml.jackson.databind.ObjectMapper;
import java.io.File;
import java.io.FileWriter;
import java.io.IOException;
import java.util.Arrays;
import java.util.Collection;
import java.util.List;

@RunWith(Parameterized.class)
public class LiveAxeCoreTest {
    private WebDriver driver;
    private String siteUrl;

    public LiveAxeCoreTest(String siteUrl) {
        this.siteUrl = siteUrl;
    }

    @Parameters
    public static Collection siteProvider() {
        return Arrays.asList(new Object[][] {
            {"https://mn.gov/mnit/about-mnit/accessibility/office-of-inaccessibility.jsp"},
            {"https://www.google.com"},
            {"https://www.bing.com"},
            {"https://www.yahoo.com"}
        });
    }

    @Before
    public void setUp() {
        WebDriverManager.chromedriver().setup();
        ChromeOptions options = new ChromeOptions();
        options.addArguments("--start-maximized");
        driver = new ChromeDriver(options);
    }

    @Test
    public void runAxeCoreScanOnSite() {
        System.out.println("Navigating to page: " + siteUrl);
        try {
            driver.get(siteUrl);
            AxeBuilder axeBuilder = new AxeBuilder();
            System.out.println("Executing axe-core accessibility scan on " + siteUrl + "...");
            Results axeResults = axeBuilder.analyze(driver);
            System.out.println("Axe-core scan completed for " + siteUrl + ".");

            ObjectMapper mapper = new ObjectMapper();
            File jsonReportDir = new File("target/a11y-json-reports");
            if (!jsonReportDir.exists()) {
                jsonReportDir.mkdirs();
            }
            String jsonFileName = siteUrl.replaceAll("[^a-zA-Z0-9.-]", "_") + "_axe_results.json";
            File jsonFile = new File(jsonReportDir, jsonFileName);
            mapper.writerWithDefaultPrettyPrinter().writeValue(jsonFile, axeResults);
            System.out.println("Axe-core JSON results saved to: " + jsonFile.getAbsolutePath());

            generateHtmlReport(axeResults, siteUrl);

            if (axeResults.getViolations().isEmpty()) {
                System.out.println("No accessibility violations found on " + siteUrl + "!");
                System.out.println("This means the page appears to be highly accessible.");
            } else {
                System.out.println("--- Accessibility Violations Found on " + siteUrl + " ---");
                System.out.println("Review these issues to improve the accessibility of the page.");
                axeResults.getViolations().forEach(violation -> {
                    System.out.println("Rule ID: " + violation.getId());
                    System.out.println("Description: " + violation.getDescription());
                    System.out.println("Help URL: " + violation.getHelpUrl());
                    System.out.println("Impact: " + violation.getImpact());
                    System.out.println("Nodes:");
                    violation.getNodes().forEach(node -> {
                        System.out.println("    HTML: " + node.getHtml());
                        System.out.println("    Target: " + node.getTarget());
                        System.out.println("    Failure Summary: " + node.getHtml());
                        System.out.println("-----");
                    });
                    System.out.println("------------------------------------");
                });
            }

            Assert.assertTrue("Accessibility violations found on " + siteUrl + ": " + axeResults.getViolations().toString(),
                axeResults.getViolations().isEmpty());
        } catch (AxeRuntimeException e) {
            System.err.println("Axe-core runtime error during analysis of " + siteUrl + ": " + e.getMessage());
            e.printStackTrace();
        } catch (IOException e) {
            System.err.println("Error saving Axe-core results to JSON or HTML for " + siteUrl + ": " + e.getMessage());
            e.printStackTrace();
        } catch (Exception e) {
            System.err.println("An unexpected error occurred while testing " + siteUrl + ": " + e.getMessage());
            e.printStackTrace();
        }
    }

    private static void generateHtmlReport(Results results, String url) throws IOException {
        StringBuilder htmlContent = new StringBuilder();
        htmlContent.append("\n\n\n\n\nAccessibility Report for ").append(url).append("\n");
        htmlContent.append("\n");
        htmlContent.append("\n\n
Accessibility Report
\n
Page Scanned: ").append(url).append("
\n
Summary
Total Violations: ").append(results.getViolations().size()).append("
Total Inapplicable: ").append(results.getInapplicable().size()).append("
Total Incomplete: ").append(results.getIncomplete().size()).append("
\n");
        if (!results.getViolations().isEmpty()) {
            htmlContent.append("
Violations (").append(results.getViolations().size()).append(")
\n\n");
            for (Rule violation : results.getViolations()) {
                String impactClass = "violation-impact-" + violation.getImpact();
                htmlContent.append("\n");
            }
            htmlContent.append("
Rule IDImpactDescriptionHTML AffectedFailure SummaryHelp Link
").append(escapeHtml(violation.getId())).append("").append(escapeHtml(violation.getImpact())).append("").append(escapeHtml(violation.getDescription())).append("\n");
                for (Node node : violation.getNodes()) {
                    htmlContent.append("
").append(escapeHtml(node.getHtml())).append("
Target: ").append(escapeHtml(node.getTarget().toString())).append("
\n");
                }
                htmlContent.append("		\n");
                for (Node node : violation.getNodes()) {
                    htmlContent.append("
").append(escapeHtml(node.getHtml())).append("
\n");
                }
                htmlContent.append("		Learn More
\n");
        } else {
            htmlContent.append("
No accessibility violations found.
\n");
        }
        htmlContent.append("\n");
        File htmlReportDir = new File("target/a11y-html-reports");
        if (!htmlReportDir.exists()) {
            htmlReportDir.mkdirs();
        }
        String htmlFileName = url.replaceAll("[^a-zA-Z0-9.-]", "_") + "_accessibility_report.html";
        File htmlFile = new File(htmlReportDir, htmlFileName);
        try (FileWriter writer = new FileWriter(htmlFile)) {
            writer.write(htmlContent.toString());
        }
        System.out.println("HTML accessibility report generated at: " + htmlFile.getAbsolutePath());
    }

    private static String escapeHtml(String text) {
        if (text == null) {
            return "";
        }
        return text.replace("&", "&")
            .replace("<", "<")
            .replace(">", ">")
            .replace("\"", """)
            .replace("'", "'");
    }

    @After
    public void tearDown() {
        if (driver != null) {
            System.out.println("Closing browser after testing " + siteUrl + ".");
            driver.quit();
        }
    }
}
